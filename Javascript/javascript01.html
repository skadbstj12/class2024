01. 변수 : 데이터 저장

자바스크립트에서 변수는 값을 지정하고 참조하는데 사용합니다. 변수를 사용하면 데이터를 저장하고 나중에 해당 데이터를 재사용 할 수 있습니다. 

{
    let x = 100;  // let x = 100; : 여기서 let 키워드를 사용하여 변수 x를 선언하고, 100이라는 숫자 값을 할당합니다.
    let y = 200;  // let y = 200; : 이 라인에서도 let을 사용해 y라는 변수를 선언하고, 200이라는 숫자 값을 할당합니다
    let z = "javascript";  // let z = "javascript"; : 이 라인에서는 z라는 변수를 선언하고, "javascript"라는 문자열 값을 할당합니다
    console.log(x); // console.log() 함수를 사용하여 변수 x의 값을 콘솔에 출력합니다. 이 경우, 100이 출력됩니다.
    console.log(y); // console.log() 함수를 사용하여 변수 y의 값을 콘솔에 출력합니다. 200이 출력됩니다.
    console.log(z); // console.log() 함수로 변수 z의 값을 콘솔에 출력합니다. 이 경우 출력되는 값은 "javascript" 문자열입니다.
}
결과 확인하기
100
200
javascript

02. 변수 : 데이터 저장 + 데이터 변경

변수는 프로그래밍에서 데이터를 저장하고 변경하는 데 사용됩니다. 이는 프로그램이 실행되는 동안 필요한 값을 저장하고, 필요에 따라 해당 값을 변경할 수 있게 해줍니다. 변수는 메모리에 할당되며, 해당 메모리 위치에는 변수에 할당된 데이터가 저장됩니다. 이를 통해 프로그램은 실행 중에 동적으로 데이터를 다룰 수 있습니다. 

{
    let x = 100; // let 키워드를 사용하여 x라는 이름의 변수를 선언하고, 그 값을 100으로 초기화합니다.
    let y = 200;  // 200;: y라는 또 다른 변수를 선언하고 200으로 초기화합니다.
    let z = "javascript"; // "javascript";: z라는 변수를 선언하고 문자열 "javascript"로 초기화합니다.

    x = 300;  // 300;: 이전에 선언된 x 변수의 값을 300으로 변경합니다.
    y = 400;  // 400;: 이전에 선언된 y 변수의 값을 400으로 변경합니다.
    z = "Jquery"  // "Jquery": 이전에 선언된 z 변수의 값을 문자열 "Jquery"로 변경합니다.

    console.log(x);  // 이 명령은 변경된 x 변수의 값을 콘솔에 출력하도록 지시합니다. 여기서 x는 300으로 변경되었으므로 콘솔에는 300이 출력됩니다.
    console.log(y);  // 변경된 y 변수의 값을 콘솔에 출력합니다. y의 값이 400으로 변경되었으므로 콘솔에는 400이 출력됩니다.
    console.log(z);  // 마지막으로, 변경된 z 변수의 값을 콘솔에 출력합니다. z는 문자열 "Jquery"로 변경되었으므로 콘솔에는 Jquery가 출력됩니다.
}
결과 확인하기
300
400
Jquery

03. 변수 : 상수 : 데이터 저장(변경&추가 X)

변수는 프로그래밍에서 사용되는 개념으로, 데이터를 저장하고 변경하며 필요한 경우 데이터를 추가할 수 있도록 해줍니다. 이는 프로그램이 실행되는 동안 필요한 값을 보관하고 조작하는 데에 중요한 역할을 합니다. 변수는 메모리 공간에 할당되며, 그 안에는 숫자, 텍스트, 논리값 등 다양한 종류의 데이터를 저장할 수 있습니다. 이를 통해 프로그램은 동적으로 데이터를 다룰 수 있게 되며, 유연성과 효율성을 높일 수 있습니다. 

{
    let x = 100;  // x라는 이름의 변수를 선언하고, 그 초기값으로 100을 할당합니다.
    let y = 200;  // y라는 이름의 변수를 선언하고, 그 초기값으로 200을 할당합니다.
    let z = "javascript";  // "javascript"; : z라는 이름의 변수를 선언하고, 그 초기값으로 문자열 "javascript"를 할당합니다.

    x += 300; // x 변수의 현재 값(100)에 300을 더하여, x는 이제 400이 됩니다.
    y += 400;  // y 변수의 현재 값(200)에 400을 더하여, y는 이제 600이 됩니다.
    z += "Jquery"  // z 변수의 현재 값에 문자열 "Jquery"를 이어붙여, z는 이제 "javascriptJquery"가 됩니다.

    console.log(x);  // 콘솔에 x 변수의 최종 값을 출력합니다. 따라서, 400이 출력됩니다.
    console.log(y);  // 콘솔에 y 변수의 최종 값을 출력합니다. 따라서, 600이 출력됩니다.
    console.log(z);  // 콘솔에 z 변수의 최종 값을 출력합니다. 따라서, "javascriptJquery"가 출력됩니다.
}
결과 확인하기
400
600
javascriptJquery

04. 변수 : 상수 : 데이터 저장(변경&추가 X)

상수는 프로그래밍에서 데이터를 저장하는데 사용되지만, 그 값을 변경할 수 없는 변수입니다. 즉, 한 번 정해진 값을 변경할 수 없습니다. 상수는 프로그램에서 변하지 않는 값을 나타내는 데 유용하며, 일반적으로 프로그램에서 고정된 값으로 사용됩니다. 

{
    let x = 100;  // 100; : x라는 이름의 변수를 선언하고, 그 초기값으로 100을 할당합니다.
    let y = 200;  // 200; : y라는 이름의 변수를 선언하고, 그 초기값으로 200을 할당합니다.
    let z = "javascript";  // z라는 이름의 변수를 선언하고, 그 초기값으로 문자열 "javascript"를 할당합니다.

    // x += 300;  // 이 줄은 주석 처리되어 있으므로, 코드 실행 시 x 변수의 값에 300을 더하는 연산은 수행되지 않습니다.
    // y += 400;  // 이 줄도 주석 처리되어 있으므로, 코드 실행 시 y 변수의 값에 400을 더하는 연산은 수행되지 않습니다.
    // z += "Jquery"  // "Jquery" : 이 줄 또한 주석 처리되어 있으므로, 코드 실행 시 z 변수의 값에 문자열 "Jquery"를 더하는 연산은 수행되지 않습니다.

    console.log(x);  // 콘솔에 x 변수의 값을 출력합니다. 주석 처리된 코드가 실행되지 않으므로, 초기에 할당된 값인 100이 출력됩니다.
    console.log(y);  // 콘솔에 y 변수의 값을 출력합니다. 마찬가지로, 초기에 할당된 값인 200이 출력됩니다.
    console.log(z);  // 콘솔에 z 변수의 값을 출력합니다. 초기에 할당된 문자열 값인 "javascript"가 출력됩니다.
}

05. 변수 : 상수 : 데이터 저장(변경&추가 X)

배열은 여러 개의 데이터를 저장하는 자료구조로, 데이터를 순서대로 저장합니다. 배열을 선언할 때는 대괄호([]) 안에 데이터를 나열하여 선언합니다. 

{
    const arr = new Array();  //  arr라는 이름의 상수에 새로운 빈 배열을 할당합니다. const 키워드를 사용함으로써 arr의 참조값은 변경될 수 없지만, 배열 내의 요소는 변경될 수 있습니다.
    arr[0] = 100;  // arr 배열의 첫 번째 요소(인덱스 0)에 숫자 100을 할당합니다.
    arr[1] = 200;  // arr 배열의 두 번째 요소(인덱스 1)에 숫자 200을 할당합니다.
    arr[2] = "javascript";  // "javascript"; : arr 배열의 세 번째 요소(인덱스 2)에 문자열 "javascript"를 할당합니다.

    console.log("05_____________");
    console.log(arr[0]);  // 콘솔에 arr 배열의 첫 번째 요소의 값을 출력합니다. 따라서, 100이 출력됩니다.
    console.log(arr[1]);  // 콘솔에 arr 배열의 두 번째 요소의 값을 출력합니다. 따라서, 200이 출력됩니다.
    console.log(arr[2]);  // 콘솔에 arr 배열의 세 번째 요소의 값을 출력합니다. 따라서, "javascript"가 출력됩니다.
}

200

06. 배열 : 데이터 저장(여러개) : 표현방법 2 : 배열선언 + 초기값 설정

배열을 선언하는 동시에 초깃값을 설정하는 방법은 배열의 선언과 함께 원하는 데이터를 대괄호([]) 안에 나열하여 초기화하는 것을 의미합니다. 이 방법은 배열을 만들 때 한 번에 초기값을 설정할 수 있어서 편리합니다. 초기값은 순서대로 배열에 저장되며, 나중에 배열 요소를 추가하거나 변경할 수 있습니다. 

{
    const arr = new Array(100, 200, "javascript");  // arr라는 이름의 상수에 새로운 배열을 할당하며, 이 배열은 세 개의 요소를 갖습니다. 첫 번째 요소로 100, 두 번째 요소로 200, 세 번째 요소로 문자열 "javascript"가 바로 할당됩니다. 
    console.log("06_____________");
    console.log(arr[0]);  // 콘솔에 arr 배열의 첫 번째 요소의 값을 출력합니다. 배열의 인덱스는 0부터 시작하므로, 여기서는 100이 출력됩니다.
    console.log(arr[1]);  // 콘솔에 arr 배열의 두 번째 요소의 값을 출력합니다. 이 경우에는 200이 출력됩니다.
    console.log(arr[2]);  // 콘솔에 arr 배열의 세 번째 요소의 값을 출력합니다. 이 경우에는 "javascript"라는 문자열이 출력됩니다.
}

07. 배열 : 데이터 저장(여러개) : 표현방법 3 : 리터럴방식 배열선언

배열을 리터럴 방식으로 선언하는 것은 배열을 생성하고 초기화할 때 사용되는 방법입니다. 이 방식에서는 대괄호([]) 안에 원하는 데이터를 나열하여 배열을 생성합니다. 

{
    const arr = [];  // arr라는 이름의 상수에 빈 배열을 할당합니다. 이때, 대괄호([])를 사용하여 배열을 생성하는 것은 new Array()를 사용하는 것과 같은 결과를 가져오지만, 더 간결하고 널리 사용되는 방법입니다.

    arr[1] = 200;  // 200; : arr 배열의 두 번째 요소(인덱스 1)에 숫자 200을 할당합니다.
    arr[2] = "javascript";  // "javascript"; : arr 배열의 세 번째 요소(인덱스 2)에 문자열 "javascript"를 할당합니다.
    
    console.log("07_____________");
    console.log(arr[0]);  // 콘솔에 arr 배열의 첫 번째 요소의 값을 출력합니다. 따라서, 100이 출력됩니다.
    console.log(arr[1]);  // 콘솔에 arr 배열의 두 번째 요소의 값을 출력합니다. 따라서, 200이 출력됩니다.
    console.log(arr[2]);  // 콘솔에 arr 배열의 세 번째 요소의 값을 출력합니다. 따라서, "javascript"라는 문자열이 출력됩니다.
}

08. 배열 : 데이터 저장(여러개) : 표현방법 4 : 리터럴방식 배열선언 + 초기값 설정 (가장 많이 사용)

배열을 리터럴 방식으로 선언하면서 동시에 초깃값을 설정하는 것은 배열을 생성하고 초기화할 때 사용되는 간편한 방법입니다. 이 방식에서는 대괄호([]) 안에 초깃값을 나열하여 배열을 선언하고 초기화합니다. 예를 들어, [1, 2, 3, 4, 5]와 같이 배열을 선언하면서 초깃값을 설정할 수 있습니다.

    console.log(arr[1]);  // 배열 arr의 두 번째 요소의 값을 콘솔에 출력합니다. 따라서 200이 출력됩니다.
    console.log(arr[2]);  // 배열 arr의 세 번째 요소의 값을 콘솔에 출력합니다. 이 때 "javascript"라는 문자열이 출력됩니다.
}

09. 배열 : 데이터 저장(여러개) : 표현방법 5 : push() 매서드를 이용한 배열 방법

push() 메서드는 배열에 새로운 요소를 추가하는 데 사용됩니다. 이 메서드를 사용하면 배열의 끝에 새로운 요소를 추가할 수 있습니다. 예를 들어, array.push(element)와 같이 사용하여 element를 배열 array의 끝에 추가할 수 있습니다. 이 방법은 배열의 길이를 자동으로 확장하며, 추가할 요소가 많은 경우 유용하게 사용됩니다. 

{
    const arr = [100, 200, "javascript"];  // 이 코드는 arr라는 상수에 3개의 요소(숫자 100, 숫자 200, 문자열 "javascript")를 가진 배열을 할당합니다.
    arr.push(400);  // push 메소드를 사용하여 배열 arr의 끝에 숫자 400을 추가합니다. push 메소드는 배열의 마지막에 하나 이상의 요소를 추가하고, 변경된 배열의 길이를 반환합니다.
    console.log("09_____________");
    console.log(arr[0]);  // 배열 arr의 첫 번째 요소(숫자 100)를 콘솔에 출력합니다.

10. 배열 : 데이터 저장(여러개) : 표현방법 6 : 다차원 배열

다차원 배열은 배열 안에 또 다른 배열을 요소로 가지는 배열을 의미합니다. 이러한 다차원 배열은 행렬이나 표와 같은 구조를 나타내는 데 유용하게 사용됩니다. 예를 들어, 2차원 배열은 배열의 배열로서 각 요소가 배열인 배열을 말합니다. 이러한 다차원 배열을 사용하면 복잡한 데이터 구조를 표현하고 다룰 수 있습니다. 

{
        const arr = [100, [200, 300, [400, 500, [600]]], ["javascript", "Jquery"]];  // [100, [200, 300, [400, 500, [600]]], ["javascript", "Jquery"]]; : 이 코드는 arr라는 상수에 복잡한 중첩 구조를 가진 배열을 할당합니다
        console.log("10_____________");
        console.log(arr[0]);  // 배열 arr의 첫 번째 요소(숫자 100)를 출력합니다.
        console.log(arr[1][0]);  // 배열 arr의 두 번째 요소(배열 [200, 300, [400, 500, [600]]])의 첫 번째 요소(숫자 200)를 출력합니다.

11. 배열 : 데이터 저장(여러개) : 표현방법 7 : 배열구조 분해 할당

1. 배열 구조 분해 할당은 배열에서 데이터를 추출하여 여러 변수에 할당하는 방법입니다. 이 방법은 배열의 각 요소를 개별 변수에 할당하는 것을 허용합니다. 대괄호([]) 안에 있는 변수들을 배열에 할당하고, 할당하려는 변수들의 순서에 맞게 배열의 값들이 할당됩니다. 이를 통해 배열의 요소들을 간단하게 추출하고 사용할 수 있습니다. 이러한 배열 구조 분해 할당은 코드를 간결하게 만들어주고, 데이터를 더 쉽게 다룰 수 있도록 도와줍니다.

2. 배열 디스트럭처링을 사용하여 다차원 배열의 값에 접근하는 예시입니다. const arr = [100, 200, [300, 400]]; : arr은 [100, 200, [300, 400]] 배열을 가리킵니다. const [a, b, [x, y]] = arr; : 배열 디스트럭처링을 사용하여 arr의 값을 a, b, [x, y]에 할당합니다. 

{
const arr = [100, 200, "javascript"];  // 이 코드는 arr이라는 상수에 숫자 100, 숫자 200, 그리고 문자열 "javascript"를 요소로 갖는 배열을 할당합니다.
const [a, b, c] = arr;  // 이 구문은 배열 arr에서 첫 번째 요소(100), 두 번째 요소(200), 그리고 세 번째 요소("javascript")를 각각 변수 a, b, c에 할당합니다
console.log("11_____________");
console.log(a);  // 변수 a의 값인 100을 콘솔에 출력합니다.
console.log(b);  // 변수 b의 값인 200을 콘솔에 출력합니다.
console.log(c);  // 변수 c의 값인 "javascript"를 콘솔에 출력합니다.
}
{

const [a, b, [x, y]] = arr2;  // arr2; : 배열 arr2의 첫 번째 요소와 두 번째 요소는 각각 변수 a와 b에 할당되며, 세 번째 요소(배열 [300, 400])는 추가적인 구조 분해를 통해 그 내부의 요소가 변수 x와 y에 각각 할당됩니다. 따라서 a는 100, b는 200, x는 300, y는 400의 값을 가지게 됩니다.
console.log(a);  // 콘솔에 변수 a의 값을 차례로 출력합니다. 출력 결과는 100이 됩니다.

console.log(y);  // 콘솔에 변수 y의 값을 차례로 출력합니다. 출력 결과는 400이 됩니다.
}

12. 배열 : 데이터 저장(여러개) : 표현방법 8 : 배열 펼침 연산자

배열 펼침 연산자는 배열을 풀어서 개별 요소로 분해하는 연산자입니다. 이 연산자는 대괄호([]) 안에서 사용되며, 배열을 쉼표로 구분된 개별 값으로 변환합니다. 이를 통해 배열을 다른 배열이나 함수의 인자로 전달할 때, 배열의 요소를 개별적인 값으로 전달할 수 있습니다. 배열 펼침 연산자는 코드를 간결하게 만들고, 배열을 다루는 여러 가지 상황에서 유용하게 사용됩니다. 

{
const arr1 = [100, 200, "javascript"];  // 이 코드는 arr1이라는 상수에 숫자 100, 숫자 200, 그리고 문자열 "javascript"를 요소로 갖는 배열을 할당합니다.
const arr2 = [300, 400, "Jquery"]; // 이 코드는 arr2라는 다른 상수에 숫자 300, 숫자 400, 그리고 문자열 "Jquery"를 요소로 갖는 또 다른 배열을 할당합니다.
const arr3 = [...arr1, 500]; // 이 구문은 스프레드 연산자 ...를 사용하여 arr1 배열의 모든 요소를 새 배열 arr3로 복사한 다음, 마지막에 숫자 500을 추가합니다. 결과적으로 arr3는 [100, 200, "javascript", 500]의 값을 갖게 됩니다.

console.log(arr1); // [100, 200, "javascript"]
console.log(...arr1) // 100200javjascript
console.log(...arr1, ...arr2); // 100200javascirpt300400jquuery
console.log(...arr3) // 100200javascirpt500
}
결과 확인하기
