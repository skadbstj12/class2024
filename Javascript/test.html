01. 1부터 100까지 출력하기(for문, while, do while)

{
for (let i = 1; i <= 100; i++) {
console.log(i)
};
}
// 1, 2, 3, 4, ....100
{
let num = 1; //초갓값 설정

while (num <= 100) {
console.log(num);
num++;
}
}
{
let num = 1;

do {
console.log(num);
num++
} while (num <= 100);
}
결과 확인하기
1~100
1~100
1~100
